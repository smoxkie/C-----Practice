#include <iostream>
using namespace std;

int main(){
    int n;
    cout << "Enter the no of rows: ";
    cin >> n;

    for(int i=n; i>=n; --i){
        for(int j=n; j>=n; --j){
            cout << "  ";
        }

        for(int j=1; j<=i; j++){
            cout << "* ";
        }
        cout << endl;
    }
    return 0;
}
